Overview

Data Loading

demographics <- fetch_dataset(

Enrollment Analysis

Total Enrollment Trends

## ============================================================
## Enrollment trends over time
## ============================================================

if ("year" %in% names(enrollment) || "school_year" %in% names(enrollment)) {

  year_col <- if ("year" %in% names(enrollment)) "year" else "school_year"

  enrollment_trend <- enrollment |>
    group_by(year = .data[[year_col]]) |>
    summarise(
      total_enrollment = sum(as.numeric(total_enrollment), na.rm = TRUE),
      .groups = "drop"
    ) |>
    arrange(year)

  ggplot(enrollment_trend, aes(x = year, y = total_enrollment, group = 1)) +
    geom_line(color = "steelblue", linewidth = 1.2) +
    geom_point(color = "steelblue", size = 3) +
    scale_y_continuous(labels = scales::comma) +
    labs(
      title = "NYC Public School Enrollment Over Time",
      subtitle = "Source: NYC DOE Open Data",
      x = "School Year",
      y = "Total Enrollment"
    ) +
    theme_minimal(base_size = 14) +
    theme(
      axis.text.x = element_text(angle = 45, hjust = 1),
      plot.title = element_text(face = "bold")
    )

} else {
  cat("Year column not found in enrollment data\n")
}

Enrollment by Grade Level

## ============================================================
## Enrollment distribution by grade level
## ============================================================

# Look for grade-related columns
grade_cols <- names(enrollment)[grep("grade|pk|k_|_k$", names(enrollment), ignore.case = TRUE)]

if (length(grade_cols) > 0) {
  cat("Grade-related columns found:\n")
  print(grade_cols)

  # Pivot to long format for visualization
  enrollment_by_grade <- enrollment |>
    select(any_of(c("dbn", "school_name", grade_cols))) |>
    pivot_longer(
      cols = any_of(grade_cols),
      names_to = "grade",
      values_to = "count"
    ) |>
    group_by(grade) |>
    summarise(total = sum(as.numeric(count), na.rm = TRUE), .groups = "drop") |>
    filter(total > 0) |>
    mutate(grade = fct_reorder(grade, total))

  ggplot(enrollment_by_grade, aes(x = total, y = grade)) +
    geom_col(fill = "darkorange", alpha = 0.8) +
    scale_x_continuous(labels = scales::comma) +
    labs(
      title = "Total Enrollment by Grade Level",
      x = "Total Students",
      y = "Grade"
    ) +
    theme_minimal(base_size = 12)

} else {
  cat("No grade-level columns found\n")
}

Demographics Analysis

Demographic Composition

## ============================================================
## Overall demographic composition
## ============================================================

# Common demographic column patterns
demo_cols <- c("asian", "black", "hispanic", "white", "multi_racial", 
               "native_american", "pacific_islander")

available_demo <- names(demographics)[names(demographics) %in% demo_cols]

if (length(available_demo) > 0) {

  demo_totals <- demographics |>
    summarise(across(all_of(available_demo), 
                     ~sum(as.numeric(.), na.rm = TRUE))) |>
    pivot_longer(everything(), names_to = "group", values_to = "count") |>
    mutate(
      pct = count / sum(count) * 100,
      group = str_to_title(str_replace_all(group, "_", " "))
    )

  ggplot(demo_totals, aes(x = "", y = pct, fill = group)) +
    geom_col(width = 1) +
    coord_polar(theta = "y") +
    geom_text(aes(label = sprintf("%.1f%%", pct)),
              position = position_stack(vjust = 0.5),
              size = 3) +
    scale_fill_viridis_d(name = "Group") +
    labs(
      title = "NYC Student Demographics",
      subtitle = "Percentage by Ethnicity"
    ) +
    theme_void(base_size = 12) +
    theme(plot.title = element_text(face = "bold", hjust = 0.5))

} else {
  cat("Demographic columns not found. Available columns:\n")
  print(names(demographics))
}

Attendance Patterns

Chronic Absenteeism

## ============================================================
## Chronic absenteeism analysis
## (Students missing 10%+ of school days)
## ============================================================

if (exists("attendance") && nrow(attendance) > 0) {

  # Look for absenteeism rate columns
  absent_cols <- names(attendance)[grep("absent|chronic|attend", 
                                        names(attendance), 
                                        ignore.case = TRUE)]

  cat("Attendance-related columns:\n")
  print(absent_cols)

  if ("chronic_absenteeism_rate" %in% names(attendance)) {

    absentee_dist <- attendance |>
      filter(!is.na(chronic_absenteeism_rate)) |>
      mutate(rate = as.numeric(chronic_absenteeism_rate))

    ggplot(absentee_dist, aes(x = rate)) +
      geom_histogram(bins = 30, fill = "coral", color = "white", alpha = 0.8) +
      geom_vline(aes(xintercept = mean(rate, na.rm = TRUE)), 
                 color = "darkred", linetype = "dashed", linewidth = 1) +
      labs(
        title = "Distribution of Chronic Absenteeism Rates",
        subtitle = "Dashed line = citywide average",
        x = "Chronic Absenteeism Rate (%)",
        y = "Number of Schools"
      ) +
      theme_minimal(base_size = 12)

  }

} else {
  cat("Attendance data not loaded\n")
}

Attendance by School Type

## ============================================================
## Compare attendance across school types
## ============================================================

if (exists("attendance") && "school_type" %in% names(attendance)) {

  attendance_by_type <- attendance |>
    group_by(school_type) |>
    summarise(
      n_schools = n(),
      avg_attendance = mean(as.numeric(attendance_rate), na.rm = TRUE),
      avg_chronic_absent = mean(as.numeric(chronic_absenteeism_rate), na.rm = TRUE),
      .groups = "drop"
    ) |>
    arrange(desc(avg_attendance))

  cat("Attendance Metrics by School Type:\n\n")
  print(attendance_by_type)

}

Summary Statistics

## ============================================================
## Generate student summary statistics
## ============================================================

cat("\n========== NYC Students Summary ==========\n\n")

if (exists("demographics")) {
  total <- sum(as.numeric(demographics$total_enrollment), na.rm = TRUE)
  cat("Total Students (from demographics):", scales::comma(total), "\n\n")
}

if (exists("enrollment")) {
  cat("Enrollment data years available:\n")
  year_col <- intersect(c("year", "school_year"), names(enrollment))[1]
  if (!is.na(year_col)) {
    print(unique(enrollment[[year_col]]))
  }
}

cat("\nData sources used:\n")
cat("- demographic_snapshot:", NYC_DATA$demographic_snapshot, "\n")
cat("- enrollment:", NYC_DATA$enrollment, "\n")
cat("- attendance:", NYC_DATA$attendance, "\n")
