#+NAME: math-proficiency

## ============================================================
## Math proficiency levels over time
## Level 3+ = proficient
## ============================================================

if (exists("math_results") && nrow(math_results) > 0) {
  
  # Look for year and proficiency columns
  year_col <- intersect(c("year", "school_year", "test_year"), names(math_results))[1]
  
  if (!is.na(year_col)) {
    
    # Calculate proficiency rate (Level 3 + Level 4)
    math_proficiency <- math_results |>
      filter(!is.na(.data[[year_col]])) |>
      group_by(year = .data[[year_col]]) |>
      summarise(
        total_tested = sum(as.numeric(number_tested), na.rm = TRUE),
        level_3_4 = sum(as.numeric(level_3_4_count), na.rm = TRUE),
        .groups = "drop"
      ) |>
      mutate(proficiency_rate = level_3_4 / total_tested * 100) |>
      arrange(year)
    
    ggplot(math_proficiency, aes(x = year, y = proficiency_rate, group = 1)) +
      geom_line(color = "#2E86AB", linewidth = 1.2) +
      geom_point(color = "#2E86AB", size = 3) +
      geom_hline(yintercept = 50, linetype = "dashed", color = "gray50") +
      scale_y_continuous(limits = c(0, 100)) +
      labs(
        title = "NYC Math Proficiency Rate (Grades 3-8)",
        subtitle = "Percentage meeting standards (Level 3+)",
        x = "Year",
        y = "Proficiency Rate (%)"
      ) +
      theme_minimal(base_size = 14) +
      theme(
        axis.text.x = element_text(angle = 45, hjust = 1),
        plot.title = element_text(face = "bold")
      )
    
  }
  
} else {
  cat("Math results not available\n")
}

** Math vs ELA Comparison

#+NAME: math-ela-comparison
#+BEGIN_SRC R :results output graphics :file output/math_ela_comparison.png
## ============================================================
## Compare Math and ELA proficiency by year
## ============================================================

if (exists("math_results") && exists("ela_results")) {
  
  # Process both datasets similarly
  process_test_data <- function(df, subject) {
    year_col <- intersect(c("year", "school_year", "test_year"), names(df))[1]
    if (is.na(year_col)) return(NULL)
    
    df |>
      group_by(year = .data[[year_col]]) |>
      summarise(
        proficiency_rate = sum(as.numeric(level_3_4_count), na.rm = TRUE) /
                          sum(as.numeric(number_tested), na.rm = TRUE) * 100,
        .groups = "drop"
      ) |>
      mutate(subject = subject)
  }
  
  math_summary <- process_test_data(math_results, "Math")
  ela_summary <- process_test_data(ela_results, "ELA")
  
  if (!is.null(math_summary) && !is.null(ela_summary)) {
    combined <- bind_rows(math_summary, ela_summary)
    
    ggplot(combined, aes(x = year, y = proficiency_rate, color = subject, group = subject)) +
      geom_line(linewidth = 1.2) +
      geom_point(size = 3) +
      scale_color_manual(values = c("Math" = "#2E86AB", "ELA" = "#A23B72")) +
      labs(
        title = "Math vs ELA Proficiency Rates",
        subtitle = "NYC Grades 3-8 State Assessments",
        x = "Year",
        y = "Proficiency Rate (%)",
        color = "Subject"
      ) +
      theme_minimal(base_size = 14) +
      theme(
        axis.text.x = element_text(angle = 45, hjust = 1),
        legend.position = "top"
      )
  }
  
}

## ============================================================

if (exists("graduation") && nrow(graduation) > 0) {
  
  grad_year_col <- intersect(c("cohort_year", "cohort", "year"), names(graduation))[1]
  grad_rate_col <- intersect(c("total_grads_pct", "graduation_rate", "grad_rate"), names(graduation))[1]
  
  if (!is.na(grad_year_col) && !is.na(grad_rate_col)) {
    
    grad_trend <- graduation |>
      group_by(year = .data[[grad_year_col]]) |>
      summarise(
        avg_grad_rate = mean(as.numeric(.data[[grad_rate_col]]), na.rm = TRUE),
        .groups = "drop"
      ) |>
      filter(!is.na(year), !is.na(avg_grad_rate)) |>
      arrange(year)
    
    ggplot(grad_trend, aes(x = year, y = avg_grad_rate, group = 1)) +
      geom_line(color = "forestgreen", linewidth = 1.2) +
      geom_point(color = "forestgreen", size = 3) +
      geom_hline(yintercept = 80, linetype = "dashed", color = "gray50") +
      annotate("text", x = min(grad_trend$year), y = 82, 
               label = "80% Goal", hjust = 0, color = "gray50") +
      scale_y_continuous(limits = c(50, 100)) +
      labs(
        title = "NYC High School Graduation Rate",

        y = "Graduation Rate (%)"

## ============================================================
## Regents exam pass rates by subject
## ============================================================

if (exists("regents") && nrow(regents) > 0) {
  
  if ("regents_exam" %in% names(regents) || "subject" %in% names(regents)) {
    
    subject_col <- if ("regents_exam" %in% names(regents)) "regents_exam" else "subject"
    
    regents_summary <- regents |>
      group_by(subject = .data[[subject_col]]) |>
      summarise(
        total_tested = sum(as.numeric(total_tested), na.rm = TRUE),
        total_passed = sum(as.numeric(total_passed_65), na.rm = TRUE),
        .groups = "drop"
      ) |>
      mutate(
        pass_rate = total_passed / total_tested * 100,
        subject = fct_reorder(subject, pass_rate)
      ) |>
      filter(!is.na(subject), total_tested > 1000)
    
    ggplot(regents_summary, aes(x = pass_rate, y = subject)) +
      geom_col(fill = "slateblue", alpha = 0.8) +
      geom_vline(xintercept = 65, linetype = "dashed", color = "gray50") +
      labs(
        title = "Regents Exam Pass Rates by Subject",
        subtitle = "Passing score = 65+",
        x = "Pass Rate (%)",
        y = NULL
      ) +
      theme_minimal(base_size = 12)
    
  }
  
} else {
  cat("Regents data not available\n")
